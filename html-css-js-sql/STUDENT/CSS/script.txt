<SCRIPT LANGUAGE="JavaScript">
// <!--
// Browser detection script by Tim Poulsen, Sr. Staff Writer, Ziff-Davis Education

// create some shorthand notations for the objects of interest
// and convert them to lower case for ease of handling
var name = navigator.appName.toLowerCase();
var ver = navigator.appVersion.toLowerCase();

// now detect
if(name == "microsoft internet explorer" && ver >="4")
  {
  // browser is IE, version 4 or greater
  if(ver.indexOf("mac") != -1)
    {
    // platform is Macintosh
    document.write('<LINK REL=StyleSheet HREF="iemac.css" TYPE="text/css">'); 
    }
  else
    {
    // platform is something else, probably Windows
    document.write('<LINK REL=StyleSheet HREF="ie.css" TYPE="text/css">');
    }
  }
else if(name == "netscape" && ver >="4")
  {
  // browser is Navigator, version 4 or greater
  if(ver.indexOf("mac") != -1)
    {
    // platform is Macintosh
    document.write('<LINK REL=StyleSheet HREF="nnmac.css" TYPE="text/css">'); 
    }
  else
    {
    // platform is something else, probably Windows
    document.write('<LINK REL=StyleSheet HREF="nn.css" TYPE="text/css">');
    }
  }
	
// -->	
</SCRIPT>
